    "import pandas as pd\n",
    "import os\n",
    "from dotenv import load_dotenv\n",
    "import requests\n",
    "import tempfile\n",
    "import re\n",
    "import datetime\n",
    "def get_csv_data(url, description = datetime.datetime.now()):\n",
    "    try:\n",
    "        load_dotenv()\n",
    "    except (FileNotFoundError,IOError) as e:\n",
    "        e.add_note('Environment variable file (.env) not found. See README.')\n",
    "        raise\n",
    "    \n",
    "    try:\n",
    "        app_id = os.environ['ESTAT_APP_ID']\n",
    "    except KeyError as e:\n",
    "        e.add_note('Environment variable ESTAT_APP_ID not found. See README.')\n",
    "        raise\n",
    "\n",
    "    if app_id == None:\n",
    "        raise OSError(\"Value of environment variable 'ESTAT_APP_ID' not found. See README.\")\n",
    "    \n",
    "    url_split = url.split(\"appId=\")\n",
    "    if len(url_split) != 2:\n",
    "        raise Exception(\"Invalid API url\")\n",
    "    url = url_split[0] + \"appId=\" + app_id + url_split[1]\n",
    "\n",
    "    # the csv has several rows of metadata terminated by a row starting with \"VALUE\".\n",
    "    # The data table starts on the next row.\n",
    "    # Put the metadata in a StringIO\n",
    "    result = {}\n",
    "    try:\n",
    "        with requests.get(url,stream=False) as estatresponse: # chunking in iter_lines doesn't work for stream=True\n",
    "            estatresponse.raise_for_status()\n",
    "\n",
    "            if estatresponse.encoding is None:\n",
    "                estatresponse.encoding = 'utf-8'\n",
    "            estatlines = estatresponse.iter_lines(chunk_size=1024, decode_unicode=True)\n",
    "            with tempfile.NamedTemporaryFile(mode='w',delete_on_close=False,encoding = 'utf-8') as fheader:\n",
    "                with tempfile.NamedTemporaryFile(mode='w',delete_on_close=False,encoding = 'utf-8') as fp:\n",
    "                    inheader = True\n",
    "                    colnum = 0\n",
    "                    for line in estatlines:\n",
    "                        if inheader == True:\n",
    "                            #count columns\n",
    "                            fields = re.split('\",\"',line)\n",
    "                            if len(fields) > colnum :\n",
    "                                colnum = len(fields)\n",
    "                            fheader.write(line)\n",
    "                            fheader.write(\"\\n\")\n",
    "                            if( line.startswith('\"VALUE\"')):\n",
    "                                inheader = False\n",
    "                                fheader.flush()\n",
    "                                fheader.seek(0)\n",
    "                        else:\n",
    "                            fp.write(line)\n",
    "                            fp.write(\"\\n\")\n",
    "                    fheader.close()\n",
    "                    fp.close()\n",
    "                    if inheader == True:\n",
    "                        errmsg = \"The stream that e-Stat returned lacks a 'VALUE' line. See temp file: \" + fheader.name\n",
    "                        raise Exception(errmsg)\n",
    "                    dfHeader = pd.read_csv(fheader.name, names = range(colnum))\n",
    "                    dfHeader = dfHeader.dropna(axis=1, how = \"all\")\n",
    "                    dfMain = pd.read_csv(fp.name)\n",
    "                    result['Description'] = description\n",
    "                    result['Header'] = dfHeader\n",
    "                    result['Main'] = dfMain\n",
    "\n",
    "    except requests.RequestException as e:\n",
    "            raise\n",
    "    \n",
    "    return result\n",
    "\n",
